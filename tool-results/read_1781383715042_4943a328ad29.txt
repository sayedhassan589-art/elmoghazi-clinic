    74→// Helper: format phone for WhatsApp (adds Egypt country code +20 if missing)
    75→const waPhone = (phone?: string) => {
    76→  if (!phone) return ''
    77→  // Convert Arabic/Indic numerals (٠-٩) to Latin (0-9) first
    78→  const latinized = phone.replace(/[٠-٩]/g, d => '٠١٢٣٤٥٦٧٨٩'.indexOf(d).toString())
    79→  const digits = latinized.replace(/[^0-9]/g, '')
    80→  if (!digits || digits.length < 3) return '' // too short to be valid
    81→  if (digits.startsWith('20')) return digits // already has country code
    82→  if (digits.startsWith('0')) return '2' + digits // starts with 0 → add 2
    83→  return '20' + digits // no prefix → add 20
    84→}

    86→// Helper: get local date string in Cairo timezone (fixes UTC offset issue)
    87→const _dateCache = new Map<string, string>()
    88→const getLocalDateStr = (date?: Date | string) => {
    89→  const d = date ? new Date(date) : new Date()
    90→  // Use full ISO including time to avoid cache collisions for dates near midnight UTC
    91→  const key = d.toISOString().slice(0, 16) // e.g. '2026-06-01T22:00'
    92→  const cached = _dateCache.get(key)
    93→  if (cached) return cached
    94→  const result = d.toLocaleDateString('en-CA', { timeZone: 'Africa/Cairo' })
    95→  _dateCache.set(key, result)
    96→  return result
    97→}

   216→function useDebouncedValue<T>(value: T, delay = 300): T {
   217→  const [debounced, setDebounced] = useState(value)
   218→  useEffect(() => {
   219→    const timer = setTimeout(() => setDebounced(value), delay)
   220→    return () => clearTimeout(timer)
   221→  }, [value, delay])
   222→  return debounced
   223→}

   225→// Arabic text normalization: remove diacritics, normalize alef/yaa/taa
   226→const normalizeArabic = (text: string): string => {
   227→  return text
   228→    .replace(/[أإآا]/g, 'ا')  // normalize alef variants
   229→    .replace(/[ة]/g, 'ه')      // taa marbuta → haa
   230→    .replace(/[ى]/g, 'ي')      // alef maqsura → yaa
   231→    .replace(/[\u064B-\u065F\u0670]/g, '') // remove diacritics/tashkeel
   232→    .replace(/\s+/g, ' ')      // normalize whitespace
   233→    .trim()
   234→    .toLowerCase()
   235→}

   237→// Fuzzy match: checks if query characters appear in order in target
   238→const fuzzyMatch = (query: string, target: string): boolean => {
   239→  const nq = normalizeArabic(query)
   240→  const nt = normalizeArabic(target)
   241→  if (nt.includes(nq)) return true // direct substring match (normalized)
   242→  // Character-by-character fuzzy match
   243→  let qi = 0
   244→  for (let ti = 0; ti < nt.length && qi < nq.length; ti++) {
   245→    if (nt[ti] === nq[qi]) qi++
   246→  }
   247→  return qi === nq.length
   248→}

   250→// Smart search: tries exact, then normalized, then fuzzy
   251→const smartSearch = (query: string, fields: (string | undefined)[]): { match: boolean; score: number } => {
   252→  if (!query.trim()) return { match: false, score: 0 }
   253→  const nq = normalizeArabic(query)
   254→  let bestScore = 0
   255→  for (const field of fields) {
   256→    if (!field) continue
   257→    const nf = normalizeArabic(field)
   258→    // Exact substring (highest score)
   259→    if (nf.includes(nq)) {
   260→      const score = nq.length / nf.length + 1 // prefer longer matches
   261→      if (score > bestScore) bestScore = score
   262→    }
   263→    // Fuzzy match (lower score)
   264→    else if (fuzzyMatch(query, field)) {
   265→      const score = 0.5
   266→      if (score > bestScore) bestScore = score
   267→    }
   268→  }
   269→  return { match: bestScore > 0, score: bestScore }
   270→}
